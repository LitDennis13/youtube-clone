cmake_minimum_required(VERSION 2.8...3.13)

project(tests VERSION 1.0 LANGUAGES C CXX)


include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/77f6bd3e75a2e7d3c67a024893417de42f58c418.zip
)
FetchContent_MakeAvailable(googletest)

enable_testing()

set(TEST_FILES
    "../JsonData/json_data.cpp"
    "../HttpImplementation/http_implementation.cpp"
    "http_implementation_request_tests.cpp"
    "json_data_tests.cpp"
)

add_compile_options(-fexceptions)
add_executable(tests ${TEST_FILES})

set_property(TARGET tests PROPERTY CXX_STANDARD 23)

target_link_libraries(
    tests
    GTest::gtest_main
)

include(GoogleTest)
gtest_discover_tests(tests)
